import React, { useEffect } from 'react';
import { Client } from 'web3-mq';
import { AppTypeEnum, Avatar, Button, Chat, ChannelList, ChannelPreviewUIComponentProps, LoginModal } from 'web3-mq-react';

import useLogin from './hooks/useLogin';

const CustomPreview = (props: ChannelPreviewUIComponentProps) => {
  const {
    active,
    avatarUrl,
    channel,
    lastMessage,
    displayTitle,
    setActiveChannel,
    unread,
    updatedAt,
  } = props;

  const onSetActiveChannel = () => {
    setActiveChannel && setActiveChannel(channel);
  };

  return (
    <div
      onClick={onSetActiveChannel}
    >
      <Avatar
        name="user"
        image={avatarUrl}
        size={30} 
      />
      <div>{displayTitle}</div>
    </div>
  )
};

export const ChannelListExample: React.FC = () => {
  const { keys, init, fastestUrl, logout, signMetaMask } = useLogin();

  const { keys, init, fastestUrl, getEthAccount, login, logout, register } = useLogin();
  
  useEffect(() => {
    init();
    document.getElementsByTagName('body')[0].setAttribute('data-theme', 'light');
  }, [])

  if (!keys) {
    return (
      <LoginModal 
        appType={AppTypeEnum.pc}
        register={register}
        login={login}
        getEthAccount={getEthAccount}
        loginBtnNode={<Button size='large' type='ghost'>login</Button>}
      />
    );
  }
  if (!fastestUrl) {
    return null;
  }

  const client = Client.getInstance(keys);

  return (
    <div id='box' style={{position: 'relative'}}>
      <Chat client={client} appType={AppTypeEnum['h5']} logout={logout}>
        <ChannelList 
          Preview={CustomPreview}
        />
      </Chat>
    </div>
  );
};
