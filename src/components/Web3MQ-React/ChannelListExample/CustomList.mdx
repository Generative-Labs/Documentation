import React, { useEffect, PropsWithChildren } from 'react';
import { Client } from 'web3-mq';
import { AppTypeEnum, Chat, ChannelList, ChannelListMessengerProps, Button, LoginModal } from 'web3-mq-react';

import useLogin from './hooks/useLogin';

const CustomList = (props: PropsWithChildren<ChannelListMessengerProps>) => {
  const {
    children,
    listRef,
    error,
    loading,
  } = props;
  if (error) {
    return <div>Load Error</div>
  };
  if (loading) {
    return <div>Loading...</div>;
  }
  return (
    <div ref={listRef} style={{height: '100%', overflowY: 'auto'}}>{children}</div>
  )
}

export const ChannelListExample: React.FC = () => {
  const { keys, init, fastestUrl, getEthAccount, login, logout, register } = useLogin();
  
  useEffect(() => {
    init();
    document.getElementsByTagName('body')[0].setAttribute('data-theme', 'light');
  }, [])

  if (!keys) {
    return (
      <LoginModal 
        appType={AppTypeEnum.pc}
        register={register}
        login={login}
        getEthAccount={getEthAccount}
        loginBtnNode={<Button size='large' type='ghost'>login</Button>}
      />
    );
  }
  if (!fastestUrl) {
    return null;
  }

  const client = Client.getInstance(keys);

  return (
    <div id='box' style={{position: 'relative'}}>
      <Chat client={client} appType={AppTypeEnum['h5']} logout={logout}>
        <ChannelList List={CustomList}/>
      </Chat>
    </div>
  );
};
