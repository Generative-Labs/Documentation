        justifyContent: 'center',
        cursor: 'pointer'
      }} onClick={() => setVisible(true)}>Open Modal</button>
      <Modal 
        visible={visible}

          <div>Basic Usage</div>

          <div>Basic Usage</div>
          <div>Basic Usage</div>
