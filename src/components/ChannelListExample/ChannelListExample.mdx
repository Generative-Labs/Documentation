```tsx 
import React, { useEffect } from 'react';
import { Client } from 'web3-mq';
import { Chat, Channel, ChannelList, AppTypeEnum } from 'web3-mq-react';

import useLogin from '../ChatExample/hooks/useLogin';

export const ChannelListExample: React.FC = () => {
  const { keys, init, fastestUrl, logout, signMetaMask } = useLogin();
  
  useEffect(() => {
    init();
  }, [])

  if (!keys) {
    return (
      <div>
        <button onClick={signMetaMask}>signMetaMask</button>
      </div>
    );
  }
  if (!fastestUrl) {
    return null;
  }

  const client = Client.getInstance(keys);

  return (
    <div id='box' style={{position: 'relative', border: '1px solid #f2f2f2',  width: '100%',height: '300px',overflow: 'auto'}}>
      <Chat client={client} appType={AppTypeEnum['h5']} logout={logout}>
        <ChannelList />
      </Chat>
    </div>
  );
};
```
